1. Creating the popup with specified element(.popupArea).

2. 페이지 불러와서 팝업으로 생성하기.

2. Load page and create it as a popup.

<script type="text/javascript">
    var popup = N().popup("page.html");
    popup.open();
</script>

3. 페이지 불러와서 드래그 가능한 팝업으로 생성하기.

3. Load page and create it as draggable popup.

<script type="text/javascript">
       var popup = N().popup({
           url: "page.html",
           title: "Title",
           width: 800,
           draggable: true
       });
       popup.open();
</script>

4. 팝업이 열릴 때마다 페이지가 다시 로드되는 팝업 생성하기.

4. Create a popup that reloads the page every time a popup is opened.

<script type="text/javascript">
       N().popup({
           url: "page.html",
           title: "Title",
           width: 800,
           closeMode: "remove"
       }).open();
</script>

5. 팝업에 데이터를 전달하고 받기.

5. Passing and receiving data in popups.

5.1. 팝업 페이지(popup.html)

5.1. Popup page(popup.html)

<!-- Style -->
<style>
    .popup-0001 .result { width: 400px; height: 300px; border: 1px solid var(--md-sys-color-outline-variant); }
</style>

<!-- View -->
<article class="popup-0001" title="onOpen Example">
    <p class="result"></p>
    <button class="btn-ok">Ok</button>
</article>

<!-- Controller -->
<script type="text/javascript">
N(".popup-0001").cont({
    init : function(view, request) {
        var caller = this.caller; // caller는 이 팝업 페이지를 열어준 N.popup 인스턴스
        N(".btn-ok", view).on("click", function(onCloseData) { // Ok 버튼에 이벤트 바인딩
            caller.close("Hello onClose."); // close 함수의 첫 번째 인자로 onCloseData 데이터 전달
        });
    },
    // 팝업이 열릴 때마다 실행될 함수
    onOpenFn : function(onOpenData) {
        N(".result", this.view).text(onOpenData);
    }
});
</script>
<!-- Style -->
<style>
    .popup-0001 .result { width: 400px; height: 300px; border: 1px solid var(--md-sys-color-outline-variant); }
</style>

<!-- View -->
<article class="popup-0001" title="onOpen Example">
    <p class="result"></p>
    <button class="btn-ok">Ok</button>
</article>

<!-- Controller -->
<script type="text/javascript">
N(".popup-0001").cont({
    init : function(view, request) {
        var caller = this.caller; // caller is the N.popup instance that opened this popup page.
        N(".btn-ok", view).on("click", function(onCloseData) { // Binds event to Ok button
            caller.close("Hello onClose."); // Passes the onCloseData data as the first argument of the close function.
        });
    },
    // This function is executed whenever a popup is opened.
    onOpenFn : function(onOpenData) {
        N(".result", this.view).text(onOpenData);
    }
});
</script>

5.2. 부모 페이지(parent.html)

5.2. Parent page(parent.html)

<!-- Style -->
<style>
    .parent-0001 .result { width: 400px; height: 300px; border: 1px solid var(--md-sys-color-outline-variant); }
</style>

<!-- View -->
<article class="parent-0001" title="onOpen Example">
    <p class="result"></p>
</article>

<!-- Controller -->
<script type="text/javascript">
N(".parent-0001").cont({
    init : function(view, request) {
        var popup = N().popup({
            url: "./popup.html",
            onOpen: "onOpenFn", // 팝업이 열릴 때마다 실행될 함수명
            onClose: function(onCloseData) { // 팝업이 닫힐 때마다 실행될 onClose 이벤트 핸들러
                N(".result", view).text(onCloseData);
            }
        });
        popup.open("Hello onOpen."); // open 함수의 첫 번째 인자로 onOpenData 데이터 전달
    }
});
</script>
<!-- Style -->
<style>
    .parent-0001 .result { width: 400px; height: 300px; border: 1px solid var(--md-sys-color-outline-variant); }
</style>

<!-- View -->
<article class="parent-0001" title="onOpen Example">
    <p class="result"></p>
</article>

<!-- Controller -->
<script type="text/javascript">
N(".parent-0001").cont({
    init : function(view, request) {
        var popup = N().popup({
            url: "./popup.html",
            onOpen: "onOpenFn", // Function name to be executed each time a popup is opened
            onClose: function(onCloseData) { // onClose event handler to be fired whenever the popup is closed
                N(".result", view).text(onCloseData);
            }
        });
        popup.open("Hello onOpen."); // Passes the onOpenData data as the first argument of the open function.
    }
});
</script>
