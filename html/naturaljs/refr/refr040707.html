var data = [{

1. 폼에 데이터 바인딩 하기.

1. Binds the data to the form.

폼의 context 요소는 table 이든 div 든 상관없습니다. context 요소 안에 데이터 객체의 프로퍼티명과 같은 id 속성을 가진 요소가 있다면 해당 요소에 값을 바인딩합니다.

The form's context element doesn't matter whether it's a table or a div. If there is an element with an id attribute equal to the property name of the data object in the context element, the value is bound to the element.

<table class="form">
    <tr>
        <th><label for="name">name</label></th>
        <td><input id="name" type="text"></td>
    </tr>
    <tr>
        <th><label for="age">age</label></th>
        <td><input id="age" type="text"></td>
    </tr>
    <tr>
        <th><label for="email">email</label></th>
        <td><input id="email" type="text"></td>
    </tr>
    <tr>
        <th>registered</th>
        <td id="registered"></td>
    </tr>
</table>

<script type="text/javascript">
    N(data).form(".form").bind(); // ==> N(data).form({ context: ".form", row: 0 }).bind();
</script>

2. 폼에 데이터 바인딩 및 데이터 포맷팅 -> 저장 확인 메시지 표시(N.alert) 후 폼의 전체 데이터 유효성 검증(N.validator) -> 폼 데이터를 서버에 전송(N.comm)

2. Binds data to the form and format the data -> Validates(N.validator) entire data in the form after displaying(N.alert) the save confirm message -> Submits(N.comm) the form data to the server.

<table class="form">
    <tr>
        <th><label for="name">name</label></th>
        <td><input id="name" type="text"></td>
    </tr>
    <tr>
        <th><label for="age">age</label></th>
        <td><input id="age" type="text" data-validate='[ ["required"], ["integer"] ]'></td>
    </tr>
    <tr>
        <th><label for="email">email</label></th>
        <td><input id="email" type="text" data-validate='[ ["email"] ]'></td>
    </tr>
    <tr>
        <th><label for="registered">registered</label></th>
        <td><input id="registered" type="text" data-format='[ ["date", 8, "date"] ]'></td>
    </tr>
</table>
<button class="btn-save">저장</button>

<script type="text/javascript">
    var form = N(data).form(".form").bind();

    N(".btn-save").on("click", function() {
        N(window).alert({ // 1. 저장 확인 메시지 표시
            msg : "저장 하시겠습니까?",
            confirm : true,
            onOk : function() {
                if(form.validate()) { // 2. 폼의 전체 데이터 유효성 검증
                    form.data(false).comm("save.json").submit(); // 3. 폼 데이터를 서버에 전송
                }
            }
        }).show();
    });
</script>

    <tr>
        <th><label for="name">name</label></th>
        <td><input id="name" type="text"></td>
    </tr>
    <tr>
        <th><label for="age">age</label></th>
        <td><input id="age" type="text" data-validate='[ ["required"], ["integer"] ]'></td>
    </tr>
    <tr>
        <th><label for="email">email</label></th>
        <td><input id="email" type="text" data-validate='[ ["email"] ]'></td>
    </tr>
    <tr>
        <th><label for="registered">registered</label></th>
        <td><input id="registered" type="text" data-format='[ ["date", 8, "date"] ]'></td>
    </tr>
</table>
<button class="btn-save">Save</button>

<script type="text/javascript">
    var form = N(data).form(".form").bind();

    N(".btn-save").on("click", function() {
        N(window).alert({ // 1. displaying the save confirm message.
            msg : "Do you want to save it?",
            confirm : true,
            onOk : function() {
                if(form.validate()) { // 2. Validates(N.validator) entire data in the form.
                    form.data(false).comm("save.json").submit(); // 3. Submits the form data to the server.
                }
            }
        }).show();
    });
</script>

3. 검색조건 블록에 N.form을 적용하여 조회 로직을 간단하게 만들기.

3. Making the search logic simply by applying N.form to search condition block.

<table class="form">
    <tr>
        <th><label for="name">name</label></th>
        <td><input id="name" type="text" data-validate='[ ["required"] ]'></td>
        <th><label for="age">age</label></th>
        <td><input id="age" type="text"></td>
    </tr>
</table>
<button class="btn-search">검색</button>

<script type="text/javascript">
    // add 메서드를 실행하면 id 속성을 갖고 있는 입력 요소들의 name 속성 값과 value 속성 값으로 data 객체를 생성합니다.
    var form = N(data).form(".form").add(); // { "name": "", "age": "" }

    N(".btn-search").on("click", function() {
        if(form.validate()) { // 1. 검색 조건 블록에 입력된 데이터를 유효성을 검사합니다
            form.data(false).comm("data.json").submit(); // 2. 폼의 데이터를 서버 파라미터로 전송합니다.
        }
    });
</script>
<table class="form">
    <tr>
        <th><label for="name">name</label></th>
        <td><input id="name" type="text" data-validate='[ ["required"] ]'></td>
        <th><label for="age">age</label></th>
        <td><input id="age" type="text"></td>
    </tr>
</table>
<button class="btn-search">Search</button>

<script type="text/javascript">
    // When the add method is executed, a data object is created with the name and value attribute's value of the input elements that have the id attribute.
    var form = N(data).form(".form").add(); // { "name": "", "age": "" }

    N(".btn-search").on("click", function() {
        if(form.validate()) { // 1. Validate the data entered in the search condition block
            form.data(false).comm("data.json").submit(); // 2. Sends the form data to server parameters
        }
    });
</script>

4. N.select와 N.form 연동 하기.

4. Linking N.select and N.form.

N.select와 N.form을 연동하여 사용할 때는 반드시 N.select를 먼저 초기화 한 다음 N.form을 초기화해야 합니다.

When using N.select in link with N.form, N.select must be initialized first and then N.form initialized.

<table class="form">
    <tr>
        <th><label for="name">name</label></th>
        <td><input id="name" type="text"></td>
    </tr>
    <tr>
        <th><label for="age">age</label></th>
        <td><input id="age" type="text"></td>
    </tr>
    <tr>
        <th><label for="gender">gender</label></th>
        <td><input id="gender" type="radio"></td>
    </tr>
    <tr>
        <th><label for="email">email</label></th>
        <td><input id="email" type="text"></td>
    </tr>
    <tr>
        <th>registered</th>
        <td id="registered"></td>
    </tr>
</table>

<script type="text/javascript">
    // 1. gender 라디오 셀렉트 생성
    N([{
         "key" : "male",
         "val" : "male"
     },{
         "key" : "female",
         "val" : "female"
    }]).select({
        context : ".form #gender",
        key : "key",
        val : "val"
    })

    // 2. 폼 데이터 바인딩
    N(data).form(".form").bind();
</script>

    <tr>
        <th><label for="name">name</label></th>
        <td><input id="name" type="text"></td>
    </tr>
    <tr>

        <td><input id="age" type="text"></td>

        context : ".form #gender",
        key : "key",
        val : "val"

    // 2. Binds the form data
    N(data).form(".form").bind();
</script>

5. N.select를 사용하지 않고 radio, checkbox 입력 요소를 N.form과 연동 하기.

5. Linking radio and checkbox input elements with N.form without using N.select.

radio와 checkbox 입력 요소의 첫 번째 요소에 name 속성 값과 같은 id 속성 값을 반드시 정의해야 합니다.

The id element value, such as the name attribute value, must be defined in the first element of the radio and checkbox input elements.

<div class="form">
    <label><input id="age" name="age" value="10"><span>10</span></label>
    <label><input name="age" value="20"><span>20</span></label>
    <label><input name="age" value="30"><span>30</span></label>
</div>
<div class="form">
    <label><input id="color" name="color" value="red"><span>red</span></label>
    <label><input name="color" value="blue"><span>blue</span></label>
    <label><input name="color" value="green"><span>green</span></label>
</div>

<script type="text/javascript">
    var data = {
        age : 10,
        color : ["red", "green"]
    };

    N(data).form(".form").bind();
</script>
